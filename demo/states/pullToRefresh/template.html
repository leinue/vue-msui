下拉刷新


<div class="content pull-to-refresh-content" v-pull-to-refresh="refresh" data-ptr-distance="55">
    <!-- 默认的下拉刷新层 -->
    <div class="pull-to-refresh-layer">
        <div class="preloader"></div>
        <div class="pull-to-refresh-arrow"></div>
    </div>
</div>


//返回promise对象的方法
refresh(){
    return new Promise((resolve,reject)=>{
        setTimeout(()=>{
            resolve('刷新了')
        },3e3)
    })
}
